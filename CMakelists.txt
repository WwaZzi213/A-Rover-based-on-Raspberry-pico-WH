cmake_minimum_required(VERSION 3.15)

include($ENV{PICO_SDK_PATH}/external/pico_sdk_import.cmake)

project(Rover_cpp C CXX ASM)
set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

pico_sdk_init()

add_executable(Rover_cpp
    main.cpp
    dhcpserver.c
)

target_include_directories(Rover_cpp PRIVATE ${CMAKE_CURRENT_LIST_DIR})

pico_enable_stdio_usb(Rover_cpp 1)
pico_enable_stdio_uart(Rover_cpp 0)

target_link_libraries(Rover_cpp
    pico_stdlib
    hardware_pwm

    # WiFi + lwIP (必须，否则没有网络栈)
    pico_cyw43_arch_lwip_threadsafe_background

    # BSD sockets（我们用它写极简HTTP）
    pico_lwip_sockets
)

pico_add_extra_outputs(Rover_cpp)
